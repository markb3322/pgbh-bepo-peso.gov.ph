  function esc(str) {
    if (!str) return '';
    return String(str).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
  }

  function updateStats() {
    var totalEvents = allEvents.length;
    var totalStudents = allEvents.reduce(function(sum, evt) { return sum + (parseInt(evt.total_students) || 0); }, 0);
    var completeCount = allEvents.filter(function(evt) { return evt.status === 'Complete'; }).length;
    var incompleteCount = allEvents.filter(function(evt) { return evt.status === 'Incomplete'; }).length;
    document.getElementById('statTotalEvents').textContent = totalEvents;
    document.getElementById('statTotalStudents').textContent = totalStudents;
    document.getElementById('statComplete').textContent = completeCount;
    document.getElementById('statIncomplete').textContent = incompleteCount;
  }

  async function loadData() {
    try {
      // Load locations for municipality dots
      var locRes = await sb.from('locations').select('*').order('municipality').order('barangay');
      if (!locRes.error) allLocations = locRes.data || [];
      console.log('✅ Locations loaded:', allLocations.length);

      // Load schools
      var schRes = await sb.from('schools').select('*').order('school_name');
      if (!schRes.error) allSchools = schRes.data || [];
      console.log('✅ Schools loaded:', allSchools.length);

      // Load events from cdsp_events
      var evtRes = await sb.from('cdsp_events').select('*').order('created_at', { ascending: false });
      if (!evtRes.error) allEvents = evtRes.data || [];
      console.log('✅ Events loaded:', allEvents.length);

      updateStats();
      drawMunicipalityDots();
      updateMapPins();
    } catch (e) { console.error('Load error:', e); }
  }

  function initMap() {
    map = L.map('boholMap', { center: BOHOL_CENTER, zoom: 10 });
    L.tileLayer('https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png', { 
      attribution: '&copy; OpenStreetMap | BEPO-PESO CDSP', maxZoom: 19 
    }).addTo(map);
    markersLayer = L.layerGroup().addTo(map);
    districtBoundariesLayer = L.layerGroup().addTo(map);
    municipalityDotsLayer = L.layerGroup().addTo(map);
    drawDistrictBoundaries();
  }

  function drawDistrictBoundaries() {
    L.polygon([

    ], { color: '#3b6ef5', fillColor: '#3b6ef5', fillOpacity: 0.06, weight: 2.5, dashArray: '10,5' })
    .addTo(districtBoundariesLayer).bindPopup('<strong style="color:#3b6ef5;">🔵 District 1</strong><br><small>Western Bohol</small>');

    L.polygon([
      [9.82,124.05],[9.90,124.06],[9.98,124.04],[10.05,124.02],[10.10,124.05],[10.18,124.15],
      [10.22,124.30],[10.22,124.50],[10.15,124.60],[10.05,124.60],[9.95,124.50],[9.88,124.38],
      [9.82,124.25],[9.82,124.05]
    ], { color: '#ec4899', fillColor: '#ec4899', fillOpacity: 0.06, weight: 2.5, dashArray: '10,5' })
    .addTo(districtBoundariesLayer).bindPopup('<strong style="color:#ec4899;">🩷 District 2</strong><br><small>Northern/Eastern Bohol</small>');

    L.polygon([
      [9.82,124.25],[9.88,124.38],[9.95,124.50],[10.05,124.60],[10.00,124.68],[9.90,124.68],
      [9.80,124.65],[9.72,124.58],[9.65,124.48],[9.60,124.38],[9.55,124.25],[9.55,124.10],
      [9.58,124.00],[9.65,123.98],[9.72,124.00],[9.80,124.02],[9.82,124.05],[9.82,124.25]
    ], { color: '#16a34a', fillColor: '#16a34a', fillOpacity: 0.06, weight: 2.5, dashArray: '10,5' })
    .addTo(districtBoundariesLayer).bindPopup('<strong style="color:#16a34a;">🟢 District 3</strong><br><small>Southern Bohol</small>');
  }

  function drawMunicipalityDots() {
    municipalityDotsLayer.clearLayers();
    var colors = { '1': '#3b6ef5', '2': '#ec4899', '3': '#16a34a' };
    var names = { '1': 'District 1', '2': 'District 2', '3': 'District 3' };
    var drawn = {};
    allLocations.forEach(function(loc) {
      if (drawn[loc.municipality]) return;
      drawn[loc.municipality] = true;
      var coords = MUNICIPALITY_COORDS[loc.municipality];
      if (!coords) return;
      var c = colors[loc.district] || '#94a3b8';
      L.circleMarker(coords, { radius: 5, fillColor: c, color: '#fff', weight: 1.5, fillOpacity: 0.9 })
        .addTo(municipalityDotsLayer)
        .bindPopup('<strong>' + esc(loc.municipality) + '</strong><br><small style="color:' + c + ';">' + (names[loc.district]||'') + '</small>');
    });
  }

  // FIXED: Get coordinates from municipality name (not location_id)
  function getCoordsFromMunicipality(municipality) {
    var coords = MUNICIPALITY_COORDS[municipality];
    if (!coords) return BOHOL_CENTER;
    return [coords[0] + (Math.random()-0.5)*0.008, coords[1] + (Math.random()-0.5)*0.008];
  }

  function updateMapPins() {
    markersLayer.clearLayers();
    allEvents.forEach(function(evt) {
      // Use municipality field directly (not location_id)
      var coords = getCoordsFromMunicipality(evt.municipality);
      if (!coords) {
        console.warn('No coords for municipality:', evt.municipality);
        return;
      }

      var school = allSchools.find(function(s) { return s.id === evt.school_id; });
      var color = evt.status === 'Complete' ? '#16a34a' : '#f59e0b';
      var icon = evt.status === 'Complete' ? '✅' : '⚠️';
      
      L.circleMarker(coords, { radius: 12, fillColor: color, color: '#fff', weight: 3, fillOpacity: 0.95 })
        .addTo(markersLayer).bindPopup(
          '<strong>' + icon + ' ' + esc(evt.event_name) + '</strong><br><hr style="margin:4px 0;">' +
          '<small>📍 ' + esc((evt.municipality||'') + ', ' + (evt.barangay||'')) + '</small><br>' +
          '<small>🏫 ' + esc(school ? school.school_name : 'N/A') + '</small><br>' +
          '<small>📅 ' + (evt.event_date ? new Date(evt.event_date+'T00:00:00').toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'}) : 'N/A') + '</small><br>' +
          '<small>🏠 ' + esc(evt.venue||'N/A') + '</small><br>' +
          '<small>👥 ' + (evt.total_students||0) + ' students</small><br>' +
          '<small>📋 <span style="color:'+color+';font-weight:600;">' + esc(evt.status||'Incomplete') + '</span></small>'
        ).on('click', function() { map.setView(coords, 14); });
    });
    console.log('📍 Pins updated:', allEvents.length, 'events on map');
  }

    console.log('Initializing CDSP Map...');
    // FIXED: Use CDSP Navbar
    if (typeof window.renderCDSPNavbar === 'function') {
      window.renderCDSPNavbar();
    }
    initMap();
    loadData();
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
