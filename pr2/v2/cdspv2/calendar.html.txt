    async function fetchEventsFromSupabase() {
        if (!supabaseClient || !isSupabaseReady) return false;
        try {
            const { data, error } = await supabaseClient.from('cdsp_events').select('*').order('event_date');
            if (error) throw error;

        return false;

            const d = new Date(ev.event_date);
            return !isNaN(d.getTime()) && d.getFullYear() === year && d.getMonth() === month;
        });
    }

        for (let i = 0; i < 42; i++) {
            let dayNumber = 0;
            let isOtherMonth = false;
            let isToday = false;
            
            if (i < startWeekday) {
                // Previous month days
                let prevDate = prevMonthDays - (startWeekday - i) + 1;
                dayNumber = prevDate;
                isOtherMonth = true;
            } else if (dayCounter <= daysInMonth) {
                // Current month days
                dayNumber = dayCounter;
                isOtherMonth = false;
                if (isCurrentMonth && dayNumber === today.getDate()) isToday = true;
                dayCounter++;
            } else {
                // Next month days
                dayNumber = dayCounter - daysInMonth;
                isOtherMonth = true;
                dayCounter++;
            }
            
            const eventsToday = (!isOtherMonth && eventsByDay[dayNumber]) ? eventsByDay[dayNumber] : [];
            let eventsHtml = '';
            
            if (eventsToday.length > 0) {
                eventsHtml = `<div class="space-y-1 mt-1">`;
                eventsToday.slice(0, 2).forEach(ev => {
                    let shortTitle = ev.event_name.length > 20 ? ev.event_name.substring(0,18)+'...' : ev.event_name;
                    let statusIcon = ev.status === 'Complete' ? '✅' : '⚠️';
                    eventsHtml += `<div class="event-badge" data-event-id="${ev.id}" title="${escapeHtml(ev.event_name)}">
                        ${statusIcon} <span class="truncate">${escapeHtml(shortTitle)}</span>
                    </div>`;
                });
                if (eventsToday.length > 2) {
                    eventsHtml += `<div class="text-[9px] sm:text-[10px] text-orange-500 text-center font-semibold">+${eventsToday.length-2} more</div>`;
                }
                eventsHtml += `</div>`;
            }
            
            let additionalClass = '';
            if (isOtherMonth) additionalClass = 'other-month';
            if (isToday) additionalClass = 'today';
            
            html += `<div class="calendar-day ${additionalClass}">
                        <div class="day-number">${dayNumber}</div>
                        ${eventsHtml}
                     </div>`;
        }
        calendarGrid.innerHTML = html;
        
        document.querySelectorAll('.event-badge[data-event-id]').forEach(el => {
            el.removeEventListener('click', badgeHandler);
            el.addEventListener('click', badgeHandler);
        });
    }
    
    function badgeHandler(e) {
        e.stopPropagation();
        const id = parseInt(this.getAttribute('data-event-id'));
        if (!isNaN(id)) showEventDetails(id);
    }
    
    async function refreshDataAndRender() {
        if (supabaseClient && isSupabaseReady) {
            await fetchEventsFromSupabase();
        }
        renderCalendar();
        updatePrintMonthSelector();
    }
    
    function applyFilters() {
        currentDisplayYear = parseInt(yearSelect.value);
        currentDisplayMonth = parseInt(monthSelect.value);
        renderCalendar();
    }
    
    function resetToToday() {
        const today = new Date();
        currentDisplayYear = today.getFullYear();
        currentDisplayMonth = today.getMonth();
        yearSelect.value = currentDisplayYear;
        monthSelect.value = currentDisplayMonth;
        renderCalendar();
    }
    
    function printCalendarWithOptions() {
        let printYear = currentDisplayYear;
        let printMonth = currentDisplayMonth;
        const printMonthChoice = printMonthSelect.value;
        if (printMonthChoice !== "current") {
            const [choiceYear, choiceMonth] = printMonthChoice.split('-').map(Number);
            if (!isNaN(choiceYear) && !isNaN(choiceMonth)) {
                printYear = choiceYear;
                printMonth = choiceMonth;
            }
        }
        
        const originalYear = currentDisplayYear;
        const originalMonth = currentDisplayMonth;
        currentDisplayYear = printYear;
        currentDisplayMonth = printMonth;
        renderCalendar();
        
        const paperSize = printPaperSize.value;
        const orientation = printOrientation.value;
        let paperCSS = '';
        if (paperSize === 'A4') paperCSS = `size: A4 ${orientation};`;
        else if (paperSize === 'SHORT') paperCSS = `size: 8.5in 11in ${orientation};`;
        else if (paperSize === 'LONG') paperCSS = `size: 8.5in 13in ${orientation};`;
        
        const style = document.createElement('style');
        style.id = 'printDynamicStyle';
        style.textContent = `@page { ${paperCSS} margin: 1.2cm; }`;
        const existingStyle = document.getElementById('printDynamicStyle');
        if (existingStyle) existingStyle.remove();
        document.head.appendChild(style);
        
        window.print();
        
        currentDisplayYear = originalYear;
        currentDisplayMonth = originalMonth;
        renderCalendar();
        style.remove();
    }
    
    function updatePrintMonthSelector() {
        printMonthSelect.innerHTML = '<option value="current">📌 Current View</option>';
        const monthsSet = new Set();
        allEvents.forEach(ev => {
            if (ev.event_date) {
                const d = new Date(ev.event_date);
                if (!isNaN(d.getTime())) {
                    const key = `${d.getFullYear()}-${d.getMonth()}`;
                    monthsSet.add(key);
                }
            }
        });
        const sortedMonths = Array.from(monthsSet).sort();
        sortedMonths.forEach(monthKey => {
            const [yearStr, monthStr] = monthKey.split('-');
            const year = parseInt(yearStr);
            const month = parseInt(monthStr);
            const monthName = new Date(year, month).toLocaleString('default', { month: 'long' });
            printMonthSelect.innerHTML += `<option value="${year}-${month}">${monthName} ${year}</option>`;
        });
    }
    
    function populateYearSelector() {
        const startYear = new Date().getFullYear() - 2;
        const endYear = startYear + 6;
        yearSelect.innerHTML = '';
        for (let y = startYear; y <= endYear; y++) {
            const opt = document.createElement('option');
            opt.value = y;
            opt.textContent = y;
            if (y === currentDisplayYear) opt.selected = true;
            yearSelect.appendChild(opt);
        }
    }
    
    function populateMonthSelector() {
        const monthNames = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December'];
        monthSelect.innerHTML = '';
        for (let i = 0; i < 12; i++) {
            const opt = document.createElement('option');
            opt.value = i;
            opt.textContent = monthNames[i];
            if (i === currentDisplayMonth) opt.selected = true;
            monthSelect.appendChild(opt);
        }
    }
    
    async function initApp() {
        populateYearSelector();
        populateMonthSelector();
        
        const userId = generateUserId();
        randomSuffixSpan.innerText = userId.split('-')[1] || "0000";
        document.getElementById('dynamicUserId').innerHTML = `CDSP-${randomSuffixSpan.innerText}`;
        updateFooterDate();
        
        await initSupabase();
        if (supabaseClient && isSupabaseReady) {
            await fetchEventsFromSupabase();
        } else {
            if (!allEvents.length) loadMockEvents();
        }
        renderCalendar();
        updatePrintMonthSelector();
        
        yearSelect.addEventListener('change', applyFilters);
        monthSelect.addEventListener('change', applyFilters);
        resetBtn.addEventListener('click', resetToToday);
        refreshBtn.addEventListener('click', () => refreshDataAndRender());
        printBtn.addEventListener('click', printCalendarWithOptions);
        closeModalBtn.addEventListener('click', closeModal);
        modalOverlay.addEventListener('click', (e) => { if (e.target === modalOverlay) closeModal(); });
    }
    
    initApp();
