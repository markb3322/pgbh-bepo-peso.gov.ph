        async function fetchProvinces() {
            try {
                const res = await fetch('https://psgc.gitlab.io/api/provinces.json');

                allProvinces = await res.json();

                generateProvinceCoords();
                populateRegionDropdown();
                return allProvinces;
            } catch (err) { console.error(err); return []; }
        }

        async function fetchAllMunicipalities() {
            try {
                const res = await fetch('https://psgc.gitlab.io/api/cities-municipalities.json');
                if (!res.ok) throw new Error();
                allMunicipalities = await res.json();
                generateMunicipalityCoords();
                return allMunicipalities;
            } catch (err) { console.error(err); return []; }
        }

        async function fetchBarangays(provinceCode) {
            if (barangaysCache.has(provinceCode)) return barangaysCache.get(provinceCode);
            try {
                const res = await fetch(`https://psgc.gitlab.io/api/provinces/${provinceCode}/barangays.json`);
                if (!res.ok) throw new Error();
                const data = await res.json();
                barangaysCache.set(provinceCode, data);
                return data;
            } catch (err) { console.error(err); return []; }
        }

        function generateProvinceCoords() {
            const regionCenters = {
                '010000000': [16.5, 121.0], '020000000': [17.5, 121.8], '030000000': [15.0, 121.0],
                '040000000': [14.0, 121.5], '050000000': [13.5, 123.5], '060000000': [11.0, 122.5],
                '070000000': [10.5, 124.0], '080000000': [11.5, 125.0], '090000000': [7.5, 122.5],
                '100000000': [8.5, 124.8], '110000000': [7.0, 125.5], '120000000': [6.5, 124.5],
                '130000000': [9.0, 125.5], '140000000': [17.0, 121.0], '150000000': [7.0, 124.0],
                '160000000': [9.0, 125.5], '170000000': [13.0, 121.0]
            };
            for (const prov of allProvinces) {
                const center = regionCenters[prov.regionCode] || PH_CENTER;
                const angle = (parseInt(prov.code.slice(0,4)) * 37) % 360;
                const radius = 1.2;
                provinceCoords.set(prov.code, [center[0] + Math.cos(angle * Math.PI/180) * radius, center[1] + Math.sin(angle * Math.PI/180) * radius]);
            }
        }

        function generateMunicipalityCoords() {
            for (const mun of allMunicipalities) {
                const provCoords = provinceCoords.get(mun.provinceCode) || PH_CENTER;
                const munisInProvince = allMunicipalities.filter(m => m.provinceCode === mun.provinceCode);
                const idx = munisInProvince.findIndex(m => m.code === mun.code);
                const angle = (idx * 137.5) * Math.PI / 180;
                const radius = 0.25 + (idx / Math.max(munisInProvince.length, 1)) * 0.2;
                municipalityCoords.set(mun.code, [provCoords[0] + Math.cos(angle) * radius, provCoords[1] + Math.sin(angle) * radius]);
            }
        }

                    case '170000000': name = 'Region IV-B (MIMAROPA)'; break;
                    case '050000000': name = 'Region V (Bicol)'; break;
                    case '060000000': name = 'Region VI (Western Visayas)'; break;
                    case '070000000': name = 'Region VII (Central Visayas)'; break;
                    case '080000000': name = 'Region VIII (Eastern Visayas)'; break;
                    case '090000000': name = 'Region IX (Zamboanga Peninsula)'; break;
                    case '100000000': name = 'Region X (Northern Mindanao)'; break;
                    case '110000000': name = 'Region XI (Davao)'; break;
                    case '120000000': name = 'Region XII (SOCCSKSARGEN)'; break;

        }

        window.onRegionChange = async function() {
            const regionCode = document.getElementById('regionSelect').value;
            const provSelect = document.getElementById('provinceSelect');
            provSelect.innerHTML = '<option value="">📍 Select Province</option>';
            document.getElementById('municipalitySelect').innerHTML = '<option value="">🏛️ Select Municipality</option>';
            if (!regionCode) return;
            const filtered = allProvinces.filter(p => p.regionCode === regionCode);
            filtered.forEach(p => {
                const opt = document.createElement('option');
                opt.value = p.code;
                opt.textContent = p.name;
                provSelect.appendChild(opt);
            });
        };

        window.onProvinceChange = async function() {
            const provinceCode = document.getElementById('provinceSelect').value;
            const muniSelect = document.getElementById('municipalitySelect');
            muniSelect.innerHTML = '<option value="">🏛️ Select Municipality</option>';
            if (!provinceCode) return;
            
            selectedProvinceCode = provinceCode;
            currentMunicipalities = allMunicipalities.filter(m => m.provinceCode === provinceCode);
            currentBarangays = await fetchBarangays(provinceCode);
            
            document.getElementById('muniCount').textContent = currentMunicipalities.length;
            document.getElementById('barangayCount').textContent = currentBarangays.length;
            
            currentMunicipalities.forEach(mun => {
                const opt = document.createElement('option');
                opt.value = mun.code;
                opt.textContent = mun.name;
                muniSelect.appendChild(opt);
            });
            
            drawProvincePins();
            drawMunicipalityPins();
            drawBarangayPins();
            
            const provCoords = provinceCoords.get(provinceCode);
            if (provCoords) map.flyTo(provCoords, 7, { duration: 1 });
        };

        window.onMunicipalityChange = function() {
            const munCode = document.getElementById('municipalitySelect').value;
            if (!munCode) return;
            const coords = municipalityCoords.get(munCode);
            if (coords) map.flyTo(coords, 11, { duration: 1 });
        };

        function drawProvincePins() {
            provincePinsLayer.clearLayers();
            if (!selectedProvinceCode) return;
            const province = allProvinces.find(p => p.code === selectedProvinceCode);
            const coords = provinceCoords.get(selectedProvinceCode);
            if (!province || !coords) return;
            const icon = L.divIcon({ html: `<div class="province-pin"><i class="fas fa-map-marker-alt"></i></div>`, iconSize: [28,28], iconAnchor: [14,23], popupAnchor: [0,-21] });
            const marker = L.marker(coords, { icon }).bindPopup(`<b style="color:#3b82f6;">🏞️ ${escapeHtml(province.name)}</b><br>${currentMunicipalities.length} municipalities<br>${currentBarangays.length} barangays`);
            marker.addTo(provincePinsLayer);
        }

        function drawMunicipalityPins() {
            municipalityPinsLayer.clearLayers();
            if (!selectedProvinceCode) return;
            for (const mun of currentMunicipalities) {
                const coords = municipalityCoords.get(mun.code);
                if (!coords) continue;
                const icon = L.divIcon({ html: `<div class="municipality-pin"><i class="fas fa-map-marker-alt"></i></div>`, iconSize: [26,26], iconAnchor: [13,21], popupAnchor: [0,-19] });
                const displayName = mun.name === 'City of Tagbilaran' ? 'Tagbilaran City' : mun.name;
                const marker = L.marker(coords, { icon }).bindPopup(`<b style="color:#7B2D8E;">🏛️ ${escapeHtml(displayName)}</b><br>${mun.isCity ? 'City' : 'Municipality'}<br>PSGC: ${mun.code}`);
                marker.addTo(municipalityPinsLayer);
            }
        }

        // IMPORTANT: Barangays are grouped under their specific municipality - not scattered across province!
        function drawBarangayPins() {
            barangayPinsLayer.clearLayers();
            if (!selectedProvinceCode || currentBarangays.length === 0) return;
            
            // Group barangays by municipality code
            const barangaysByMun = new Map();
            for (const brgy of currentBarangays) {
                if (!barangaysByMun.has(brgy.municipalityCode)) {
                    barangaysByMun.set(brgy.municipalityCode, []);
                }
                barangaysByMun.get(brgy.municipalityCode).push(brgy);
            }
            
            // For each municipality, place its barangays around its center
            for (const [munCode, barangays] of barangaysByMun) {
                const mun = allMunicipalities.find(m => m.code === munCode);
                if (!mun) continue;
                
                const center = municipalityCoords.get(munCode);
                if (!center) continue;
                
                const displayName = mun.name === 'City of Tagbilaran' ? 'Tagbilaran City' : mun.name;
                const maxToShow = Math.min(barangays.length, 60); // Limit for performance
                
                for (let i = 0; i < maxToShow; i++) {
                    const brgy = barangays[i];
                    // Use golden angle for even circular distribution around municipality center
                    const goldenAngle = i * 137.5 * Math.PI / 180;
                    // Radius increases with index, creating a nice cluster around the center
                    const radius = 0.008 + (i / maxToShow) * 0.014;
                    const lat = center[0] + Math.cos(goldenAngle) * radius;
                    const lng = center[1] + Math.sin(goldenAngle) * radius;
                    
                    const icon = L.divIcon({ html: `<div class="barangay-pin"></div>`, iconSize: [8,8], iconAnchor: [4,4], popupAnchor: [0, -6] });
                    const marker = L.marker([lat, lng], { icon }).bindPopup(`
                        <b style="color:#f97316;">📍 ${escapeHtml(brgy.name)}</b><br>
                        ${escapeHtml(displayName)}<br>
                        <small>PSGC: ${brgy.code}</small>
                    `);
                    marker.addTo(barangayPinsLayer);
                }
            }
        }

        function escapeHtml(str) {
            if (!str) return '';
            return str.replace(/[&<>]/g, m => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;' }[m]));
        }

        function updateStats() {
            document.getElementById('statTotalEvents').textContent = allEvents.length;
            const total = allEvents.reduce((s, e) => s + (parseInt(e.total_students) || 0), 0);
            document.getElementById('statTotalStudents').textContent = total;
            const comp = allEvents.filter(e => e.status === 'Complete').length;
            const inc = allEvents.filter(e => e.status === 'Incomplete').length;
            document.getElementById('statComplete').textContent = comp;
            document.getElementById('statIncomplete').textContent = inc;
            document.getElementById('completeBadge').textContent = comp;
            document.getElementById('incompleteBadge').textContent = inc;
        }

        function initMap() {
            map = L.map('phMap').setView(PH_CENTER, 6);
            currentTileLayer = tileLayers.satellite;
            currentTileLayer.addTo(map);

            completeEventsLayer = L.layerGroup().addTo(map);
            incompleteEventsLayer = L.layerGroup().addTo(map);
            provincePinsLayer = L.layerGroup().addTo(map);
            municipalityPinsLayer = L.layerGroup().addTo(map);
            barangayPinsLayer = L.layerGroup().addTo(map);

            Promise.all([fetchProvinces(), fetchAllMunicipalities()]).then(() => loadEvents());
        }

        window.setMapView = function(type) {
            if (currentTileLayer) map.removeLayer(currentTileLayer);
            currentTileLayer = tileLayers[type];
            currentTileLayer.addTo(map);
            const sat = document.getElementById('satelliteBtn');
            const str = document.getElementById('streetBtn');
            if (type === 'satellite') {
                sat.className = 'map-toggle-btn px-3 py-1.5 text-[11px] font-medium rounded-md bg-blue-600 text-white shadow-md';
                str.className = 'map-toggle-btn px-3 py-1.5 text-[11px] font-medium rounded-md bg-slate-700 text-slate-300';
            } else {
                str.className = 'map-toggle-btn px-3 py-1.5 text-[11px] font-medium rounded-md bg-blue-600 text-white shadow-md';
                sat.className = 'map-toggle-btn px-3 py-1.5 text-[11px] font-medium rounded-md bg-slate-700 text-slate-300';
            }
        };

        async function loadEvents() {
            try {
                const { data } = await sb.from('cdsp_events').select('*');
                allEvents = data || [];
                drawEvents();
                updateStats();
            } catch(e) { console.error(e); }
        }

        function drawEvents() {
            completeEventsLayer.clearLayers();
            incompleteEventsLayer.clearLayers();
            const bohol = [9.83, 124.22];
            allEvents.forEach((e, i) => {
                const angle = i * 137.5 * Math.PI / 180;
                const pos = [bohol[0] + Math.cos(angle) * 0.035, bohol[1] + Math.sin(angle) * 0.035];
                const isComp = e.status === 'Complete';
                const icon = L.divIcon({ html: `<div class="${isComp ? 'event-complete-pin' : 'event-incomplete-pin'}"><i class="fas ${isComp ? 'fa-check' : 'fa-exclamation'}" style="color:white; font-size:11px;"></i></div>`, iconSize: [24,24], iconAnchor: [12,12] });
                L.marker(pos, { icon }).bindPopup(`<b>${isComp ? '✅' : '⚠️'} ${escapeHtml(e.event_name)}</b><br>📍 ${e.municipality}, Bohol<br>👥 ${e.total_students || 0} students`).addTo(isComp ? completeEventsLayer : incompleteEventsLayer);
            });
        }

        window.toggleEventLayer = function(type) {
            const layer = type === 'complete' ? completeEventsLayer : incompleteEventsLayer;
            const show = type === 'complete' ? (showComplete = !showComplete) : (showIncomplete = !showIncomplete);
            const toggle = document.getElementById(type === 'complete' ? 'toggleComplete' : 'toggleIncomplete');
            if (show) { map.addLayer(layer); toggle.classList.add('active'); toggle.classList.remove('inactive'); }
            else { map.removeLayer(layer); toggle.classList.add('inactive'); toggle.classList.remove('active'); }
        };
        window.toggleProvinceLayer = function() {
            showProvinces = !showProvinces;
            const t = document.getElementById('toggleProvinces');
            if (showProvinces) { map.addLayer(provincePinsLayer); t.classList.add('active'); t.classList.remove('inactive'); }
            else { map.removeLayer(provincePinsLayer); t.classList.add('inactive'); t.classList.remove('active'); }
        };
        window.toggleMunicipalityLayer = function() {
            showMunicipalities = !showMunicipalities;
            const t = document.getElementById('toggleMunicipalities');
            if (showMunicipalities) { map.addLayer(municipalityPinsLayer); t.classList.add('active'); t.classList.remove('inactive'); }
            else { map.removeLayer(municipalityPinsLayer); t.classList.add('inactive'); t.classList.remove('active'); }
        };
        window.toggleBarangayLayer = function() {
            showBarangays = !showBarangays;
            const t = document.getElementById('toggleBarangays');
            if (showBarangays) { map.addLayer(barangayPinsLayer); t.classList.add('active'); t.classList.remove('inactive'); }
            else { map.removeLayer(barangayPinsLayer); t.classList.add('inactive'); t.classList.remove('active'); }
        };

        initMap();
    });
